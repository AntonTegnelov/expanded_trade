set_trade_tab_routes = {
	scope = character
	effect = {
		set_variable = {
			name = trade_window_tab
			value = "routes"
		}
	}
}

set_trade_tab_goods = {
	scope = character
	effect = {
		set_variable = {
			name = trade_window_tab
			value = "goods"
		}
	}
}

		}

is_trade_tab_routes = {
	scope = character
	is_shown = {
		has_variable = trade_window_tab
		var:trade_window_tab = "routes"
	}
}
