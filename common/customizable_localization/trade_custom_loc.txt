trade_income_value = {
	type = character
	text = {
		trigger = {
			has_variable = monthly_trade_income
		}
		localization_key = TRADE_INCOME_VALUE_WITH_AMOUNT
		scope = this
	}
	text = {
		trigger = {
			NOT = {
				has_variable = monthly_trade_income
			}
		}
		localization_key = TRADE_INCOME_VALUE_NONE
		scope = this
	}
}
